% Section styling -- full-width heading (not in the margin), unnumbered
% (headings carry the .unnumbered class in cv.qmd/cv-sections.lua).
% titlesec is not used here: memoir warns against it, and its non-
% [leftmargin] shapes silently fell back to memoir's own (tiny) section
% style. \sffamily (Lato) is avoided for headings too -- at larger sizes
% it was rendering far smaller than \fontsize requested, for reasons that
% didn't repay further digging; bold Crimson Text in Cornell red reads
% clearly and keeps a single font family in headings.
\setsecheadstyle{\color{CornellRed}\bfseries\fontsize{14}{17}\selectfont}
\setbeforesecskip{2.5ex plus 1ex minus .2ex}
\setaftersecskip{1.5ex plus .2ex}

\setsubsecheadstyle{\itshape\normalsize}
\setbeforesubsecskip{2ex plus .5ex}
\setaftersubsecskip{1ex}
